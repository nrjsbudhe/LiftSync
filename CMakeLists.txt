cmake_minimum_required(VERSION 3.10)

# Set the project name
project(LiftSync)

# Set the C++ standard

# Add include directories

    src/main.cpp
    src/Elevator.cpp
    src/ElevatorSystem.cpp
    src/RequestHandler.cpp
)
